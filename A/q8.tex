\begin{problem}{8}
  找到一个$\R^2\rightarrow\R $函数$\varphi$,且对于
  任意的$a\in\R$ 和 $V\in\R^2$ 都满足
  \begin{displaymath}
    \varphi(av)=a \varphi(v)
  \end{displaymath}
  并且$\varphi$不是线性的.
\end{problem}

\begin{proof}
  找到如下函数
  \begin{displaymath}
    \varphi = \ln(xy) ,\ (x,y)\in \R^2
  \end{displaymath}
  则可有
  \begin{displaymath}
    \varphi(av) = \varphi[(ax,ay)] = a\ln(xy) = a \varphi(v)
  \end{displaymath}
  但是$\varphi$ 不满足
  \begin{displaymath}
    \varphi(\nu+\omega) \neq \varphi(\nu) +\varphi(\omega)
  \end{displaymath}
  故$\varphi$不是线性的.
  
\end{proof}
