\begin{problem}{7}
  证明如果$\dim V = 1$ 并且 $T\in \Hom(V,V)$,
  存在$\lambda\in \F$ 使得对于任意的$v\in V$
  都有$Tv=\lambda v$.
\end{problem}

\begin{proof}
  因为$\dim V = 1$，所以$V$的基为单向量,设为$e$
  则存在$\alpha,\lambda$使得下式成立
  \begin{displaymath}
    Tv=T(\alpha e)=\alpha Te =\alpha \lambda e = \lambda v
  \end{displaymath}
  其中$\lambda$即为所需。原命题证明完毕
  
\end{proof}
